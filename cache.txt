server:
    depends_on:
      - db
    container_name: server
    ports:
      - "8080:80"
    build:
      context: ./server
      dockerfile: Dockerfile
    
    environment:
      - ConnectionStrings__DefaultConnection=User ID=postgres;Password=postgres;Server=db;Port=5432;Database=DBTodo;Pooling=true;
      - ASPNETCORE_URLS=http://+:80
    networks:
      - dev

  db:
    image: postgres:latest
    container_name: postgres
    environment:
      POSTGRES_USER: postgres
      POSTGRES_PASSWORD: postgres
      POSTGRES_DB: todo
    ports:
      - "5433:5432"
    restart: always
    volumes:
      - app_data:/var/lib/postgres/data
    networks:
      - dev
